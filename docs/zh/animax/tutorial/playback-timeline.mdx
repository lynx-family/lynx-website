当只需要播放动画中的一段内容时，可以设置 `start-frame` 和 `end-frame`。`speed` 用于控制播放速度，`progress` 用于设置初始进度，`loop-count` 和 `bindrepeat` / `bindcompletion` 可以让界面感知重复播放和播放完成。

```tsx {5-11}
<animax-view
  src={PLAYBACK_PROPS_LOTTIE_SRC}
  autoplay={true}
  start-frame={24}
  end-frame={144}
  auto-reverse={true}
  speed={speed}
  progress={progress}
  loop-count={2}
  bindrepeat={handleRepeat}
  bindcompletion={handleCompletion}
/>
```

`getDuration`、`getCurrentFrame` 和 `isAnimating` 会通过 UI Method 的回调返回结果。和播放控制一样，这些方法也应该在 `ready` 之后调用。

```tsx {1-15,20-23}
function invokeQuery(method: QueryMethod, onResult: (value: string) => void) {
  lynx
    .createSelectorQuery()
    .select(`#${ANIMAX_VIEW_ID}`)
    .invoke({
      method,
      success(res) {
        onResult(formatQueryResult(method, res));
      },
    })
    .exec();
}

const queryCurrentFrame = () => {
  invokeQuery('getCurrentFrame', setFrame);
};

<animax-view
  id={ANIMAX_VIEW_ID}
  src={METHODS_QUERY_LOTTIE_SRC}
  autoplay={true}
  bindready={handleReady}
/>;
```
